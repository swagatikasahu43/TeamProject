//A Simple Java Program to Add multiply Numbers. 



class MultiplyTwoNumbers        // Declaration of class
{
{
   public static void main(String args[]) //Main Function
   {
      int x, y, z;      // Variable Declaration
      x = 20;           //Assigning 10 to the variable x.
      y = 20;           //Assigning 20 to the variable y.
      z = x * y;        //Expression to Add multiply variables x, y and save the result to z
      System.out.println("Multiplication of x and y = "+z); //This line output the value of z on the Screen
   }
}
}
